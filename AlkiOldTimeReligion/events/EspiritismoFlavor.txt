country_event = { #A Faith of Many Parts
	id = espiritismo_events.1
	title = espiritismo_events.1.t
	desc = espiritismo_events.1.d
	picture = RELIGION_eventPicture
	
	trigger = {
		religion = espiritismo
		NOT = {
			has_country_flag = has_a_cult_unlocked
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = espiritismo_events.1.a #Interesting
		if = {
			limit = {
				culture_group = cubano
			}	
			unlock_cult = santeria_denom
			unlock_cult = palo_denom
			unlock_cult = abakua_denom
		}
		else_if = {
			limit = {
				culture_group = hispaniolan
			}
			unlock_cult = divisiones_denom
			unlock_cult = vodou_denom
			unlock_cult = orishan_denom
		}
		else_if = {
			limit = {
				culture_group = antillian
			}
			unlock_cult = rastafari_denom
			unlock_cult = hijos_de_dios_denom
			unlock_cult = cemi_denom
		}
		else_if = {
			limit = {
				culture_group = amazonian
			}
			unlock_cult = winti_denom
			unlock_cult = lioncero_denom
			unlock_cult = tambor_de_mina_denom
		}
		else_if = {
			limit = {
				OR = {
					culture_group = neomalay
					culture_group = neopatagonian
				}
			}
			unlock_cult = candomble_denom
			unlock_cult = umbanda_denom
			unlock_cult = anitism_denom
		}
		else_if = {
			limit = {
				capital_scope = {
					OR = {
						continent = pablonia
						continent = akilineq
					}
				}
			}
			unlock_cult = yorubite_denom
			unlock_cult = zhongish_denom
		}
		else_if = {
			limit = {
				OR = {
					technology_group = western
					technology_group = eastern
				}
			}
			unlock_cult = hoodoo_denom
		}
		else_if = {
			limit = {
				technology_group = chinese
			}
			unlock_cult = shinteria_denom
		}
		else = {
			unlock_cult = santa_muerte_denom
		}
		custom_tooltip = how_to_get_cults_explained.tt
	}
}
